```typescript
// src/lib/team-stats.ts

/** 스탯 값(0~99)에 따른 색상 등급 */
export type StatColorTier = "gold" | "green" | "blue" | "gray";

export function getStatColorTier(value: number): StatColorTier {
  if (value >= 85) return "gold";
  if (value >= 70) return "green";
  if (value >= 50) return "blue";
  return "gray";
}

/** 바 gradient CSS 클래스 (세로용) */
export const BAR_GRADIENT: Record<StatColorTier, string> = {
  gold: "from-[#8B6914] to-[#D4A853]",
  green: "from-[#166534] to-[#4ADE80]",
  blue: "from-[#1e40af] to-[#60A5FA]",
  gray: "from-[#3f3f46] to-[#71717A]",
};

/** 텍스트 색상 */
export const STAT_TEXT_COLOR: Record<StatColorTier, string> = {
  gold: "text-accent",
  green: "text-green",
  blue: "text-blue",
  gray: "text-text-3",
};
```

```typescript
/** OVR 계산: ATK×0.3 + MID×0.3 + DEF×0.25 + GK×0.15 */
export function calcTeamOvr(stats: { atk: number; mid: number; def: number; gk: number }): number {
  return Math.round(stats.atk * 0.3 + stats.mid * 0.3 + stats.def * 0.25 + stats.gk * 0.15);
}
```

```typescript
/** 팀명에서 로고 이니셜 추출 */
export function getTeamInitials(name: string): string {
  // 영문 대문자 2글자 이상 약어 우선 (FC, SC, JR 등)
  const abbr = name.match(/[A-Z]{2,}/);
  if (abbr) return abbr[0].slice(0, 2);
  // 없으면 팀명 첫 2글자
  return name.replace(/\s+/g, "").slice(0, 2);
}
```

```tsx
// src/components/team/TeamCardBars.tsx
"use client";

import { getStatColorTier, BAR_GRADIENT } from "@/lib/team-stats";

interface TeamCardBarsProps {
  atk: number;
  mid: number;
  def: number;
  gk: number;
  /** 세로 바 높이 (px) */
  height?: number;
  /** 바 두께 (px) */
  barWidth?: number;
}

const LINES: { key: "atk" | "mid" | "def" | "gk"; label: string }[] = [
  { key: "atk", label: "공격" },
  { key: "mid", label: "미드필드" },
  { key: "def", label: "수비" },
  { key: "gk", label: "골키퍼" },
];

export default function TeamCardBars({
  atk, mid, def, gk,
  height = 32,
  barWidth = 6,
}: TeamCardBarsProps) {
  const stats = { atk, mid, def, gk };

  return (
    <div
      className="flex items-end gap-[3px]"
      style={{ height }}
      role="img"
      aria-label={`공격 ${atk}, 미드필드 ${mid}, 수비 ${def}, 골키퍼 ${gk}`}
    >
      {LINES.map(({ key, label }) => {
        const val = stats[key];
        const tier = getStatColorTier(val);
        return (
          <div
            key={key}
            className={`rounded-[3px] bg-gradient-to-t ${BAR_GRADIENT[tier]}`}
            style={{
              width: barWidth,
              height: `${Math.max(val, 4)}%`,
              minHeight: 4,
            }}
            title={`${label} ${val}`}
            aria-hidden="true"
          />
        );
      })}
    </div>
  );
}
```

```tsx
// src/components/team/TeamCard.tsx
"use client";

import Link from "next/link";
import Image from "next/image";
import TeamCardBars from "./TeamCardBars";
import { getStatColorTier, STAT_TEXT_COLOR, getTeamInitials } from "@/lib/team-stats";

export interface TeamCardStats {
  ovr: number;
  atk: number;
  mid: number;
  def: number;
  gk: number;
}

export interface TeamCardData {
  id: string;
  name: string;
  logoUrl?: string | null;
  ageGroup?: string;
  city?: string | null;
  memberCount: number;
}

export interface TeamCardProps {
  team: TeamCardData;
  stats: TeamCardStats;
  rank?: number;
  tags?: string[];
  variant: "compact" | "full" | "inline";
}

export default function TeamCard({ team, stats, rank, tags, variant }: TeamCardProps) {
  if (variant === "full") return <TeamCardFull team={team} stats={stats} tags={tags} />;
  if (variant === "inline") return <TeamCardInline team={team} stats={stats} />;
  return <TeamCardCompact team={team} stats={stats} rank={rank} />;
}

/* ── Logo fallback ── */
function TeamLogo({
  name, logoUrl, size,
}: {
  name: string; logoUrl?: string | null; size: number;
}) {
  const radius = size <= 32 ? "rounded-lg" : size <= 40 ? "rounded-[10px]" : "rounded-[14px]";

  if (logoUrl) {
    return (
      <div className={`relative overflow-hidden ${radius} shrink-0`} style={{ width: size, height: size }}>
        <Image src={logoUrl} alt={name} fill sizes={`${size}px`} className="object-cover" />
      </div>
    );
  }

  return (
    <div
      className={`flex items-center justify-center ${radius} shrink-0`}
      style={{
        width: size,
        height: size,
        background: "linear-gradient(135deg, #24242A, #2C2C30)",
        border: "1px solid rgba(255,255,255,0.08)",
      }}
    >
      <span
        className="font-stat font-bold text-accent"
        style={{ fontSize: size * 0.42 }}
      >
        {getTeamInitials(name)}
      </span>
    </div>
  );
}

/* ── Meta line ── */
function MetaLine({ team }: { team: TeamCardData }) {
  const parts: string[] = [];
  if (team.ageGroup) parts.push(team.ageGroup);
  if (team.city) parts.push(team.city);
  parts.push(`${team.memberCount}명`);
  return <span className="text-[10px] text-text-3">{parts.join(" · ")}</span>;
}

/* ── Compact ── */
function TeamCardCompact({
  team, stats, rank,
}: {
  team: TeamCardData; stats: TeamCardStats; rank?: number;
}) {
  const isFirst = rank === 1;

  return (
    <Link
      href={`/team/${team.id}`}
      className={`
        relative flex items-center gap-3 rounded-[12px] p-[14px_16px]
        transition-transform active:scale-[0.985]
        ${isFirst
          ? "border border-accent/20 shadow-[0_2px_12px_rgba(212,168,83,0.08)]"
          : "border border-white/[0.07] shadow-[0_2px_8px_rgba(0,0,0,0.4)]"
        }
      `}
      style={{ background: "var(--color-card)" }}
    >
      {/* accent top line */}
      <div
        className="pointer-events-none absolute inset-x-0 top-0 rounded-t-[12px]"
        style={{
          height: isFirst ? 2 : 1.5,
          background: isFirst
            ? "linear-gradient(90deg, transparent, rgba(245,215,142,0.6), rgba(212,168,83,0.7), rgba(245,215,142,0.6), transparent)"
            : "linear-gradient(90deg, transparent, rgba(212,168,83,0.3), transparent)",
        }}
      />

      {/* rank */}
      {rank != null && (
        <span
          className={`w-5 text-center font-stat text-[14px] font-semibold shrink-0 ${
            rank === 1 ? "text-accent" : rank <= 3 ? "text-text-1" : "text-text-3"
          }`}
        >
          {rank}
        </span>
      )}

      <TeamLogo name={team.name} logoUrl={team.logoUrl} size={40} />

      <div className="min-w-0 flex-1">
        <div className="truncate text-[14px] font-bold text-text-1">{team.name}</div>
        <MetaLine team={team} />
      </div>

      {/* OVR */}
      <span
        className="shrink-0 font-stat text-[28px] font-bold leading-none"
        style={{
          background: "linear-gradient(180deg, #F5D78E, #D4A853)",
          WebkitBackgroundClip: "text",
          WebkitTextFillColor: "transparent",
        }}
      >
        {stats.ovr}
      </span>

      <TeamCardBars atk={stats.atk} mid={stats.mid} def={stats.def} gk={stats.gk} />
    </Link>
  );
}
```
